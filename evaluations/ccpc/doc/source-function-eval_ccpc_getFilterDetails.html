331:     /**
332:       * eval_ccpc_exploreQuery_operatorAction - Teste une égalité
333:       *
334:       * @category : eval_ccpc_functions
335:       * @param int $value1 Terme de gauche
336:       * @param int $value2 Terme de droite
337:       * @param string $operator Opérateur parmis : <=, <, ==, >, >=, !=
338:       * @return boolean TRUE sur l'égalité est vérifiée, FALSE si elle n'est pas vérifiée
339:       * 
340:       * @Author Ali Bellamine
341:       *
342:       */
343:     
344:     function eval_ccpc_exploreQuery_operatorAction ($value1, $value2, $operator) 
345:     {
346:         if ($operator == '<=') { return ($value1 <= $value2); }
347:         if ($operator == '<') { return ($value1 < $value2); }
348:         if ($operator == '==' || $operator == '=') { return ($value1 == $value2); }
349:         if ($operator == '>') { return ($value1 > $value2); }
350:         if ($operator == '>=') { return ($value1 >= $value2); }
351:         if ($operator == '!=') { return ($value1 != $value2); }     
352:     }
